    /**
 * 斐波那契数列
 * @param {Number} n 
 * @returns {*} dp数组
 */
function fibonacci(n) {
    // 动态规划
    // 1.初始化 dp 数组
    let dp = []
    dp[0] = 1
    dp[1] = 1
    // 2.状态转移
    for (let i = 2; i < n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2]
    }
    return dp[n - 1]
}

 */

    // 动态规划
    // 初始化 dp 数组
    let dp = []
    // 一级台阶
    dp[0] = 1
    // 二级台阶

    /**
 * 最小花费爬楼梯
 * @param {Array} cost 
 * @returns {Number}
 */
function minCostClimbStairs(cost) {
    // 动态规划

    // 1.初始化 dp 数组
    let dp = []
    // 没有楼梯
    dp[0] = 0
    // 一阶楼梯
    dp[1] = 0
    // 2.状态转移
    for (let i = 2; i <= cost.length; i++) {
        dp[i] = Math.min(dp[i - 1] + cost[i - 1], dp[i - 2] + cost[i - 2])
    }
    return dp[cost.length]
}

    # 65.最长公共子序列（二）

    /**
 * 最长公共子序列（二）
 * @param {String} s1 
 * @param {String} s2 
 * @return {String}
 */
function longestCommonString(s1, s2) {
    // 二维动态规划
    let m = s1.length, n = s2.length
    let dp = new Array(m + 1).fill(0).map(() => new Array(n + 1).fill(0))
    for (let i = 1; i <= m; i++) {
        let c1 = s1[i - 1]
        for (let j = 1; j <= n; j++) {
            let c2 = s2[j - 1]
            if (c1 === c2) {
                dp[i][j] = dp[i - 1][j - 1] + 1
            } else {
                dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1])
            }
        }
    }
    return dp[m][n]
}

    # 66.最长公共子串

    /**
 * 最长公共子串
 * @param {String} str1 
 * @param {String} str2 
 * @returns {String}
 */
function longestCommonString(str1, str2) {
    let max = 0
    const map = new Map()
    const dp = new Array(str1.length + 1).fill(0).map(() => new Array(str2.length + 1).fill(0))
    for (let i = 1; i <= str1.length; i++) {
        for (let j = 1; j <= str2.length; j++) {
            if (str1[i - 1] === str2[j - 1]) {
                // 最大子串位置
                dp[i][j] = dp[i - 1][j - 1] + 1
                // 最大子串的终点位置(长度)
                max = Math.max(max, dp[i][j])
                // 避免重复
                if (!map.has(max)) {
                    map.set(max, i)
                }
            }
        }
    }
    let startIndex = map.get(max) - max
    let endIndex = map.get(max)
    return str1.substring(startIndex, endIndex)
}

    # 67.不同路径的数目（一）

    /**
 * 不同路径的数目（一）
 * @param {Number} m x坐标
 * @param {Number} n y坐标
 * @returns {Number} 路径和
 */
function uniquePaths(m, n) {
    // 1.初始化 dp 数组
    let dp = new Array(m).fill(0).map(() => new Array(n).fill(0))
    // 如果只有一列，路径和为1
    for (let i = 0; i < m; i++) {
        dp[i][0] = 1
    }
    // 如果只有一行，路径和为1
    for (let j = 0; j < n; j++) {
        dp[0][j] = 1
    }

    // 2.状态转移
    for (let i = 1; i < m; i++) {
        for (let j = 1; j < n; j++) {
            dp[i][j] = dp[i - 1][j] + dp[i][j - 1]
        }
    }

    return dp[m - 1][n - 1]
}

    # 69.把数字翻译成字符串

    // @ts-nocheck
/**
 * 把数字翻译成字符串
 * @param {String} nums 
 * @returns {Number}
 */
 function translate(nums) {
    // 动态规划

    // 1.初始化 dp 数组
    let len = nums.length;
    let dp = new Array(len);
    dp[0] = 1;

    // 2.状态转移
    for (let i = 1; i < len; i++) {
        // 例如 i == 1， 取 0 1 两位字符串转数字
        let twoNumber = parseInt(nums.slice(i - 1, i + 1));
        if (i === len - 1 && nums[i] === 0 && twoNumber > 26) {
            // 2.1若遍历到最后一位， 当前为0， 并且当前两位 > 26, 不能转码
            dp[i] = 0;
        } else if (nums[i] === 0 || nums[i - 1] === 0 || twoNumber > 26) {
            // 2.2当前值为0，或者前一个值为0，或者当前两位 > 26,和前面一位的转码次数同
            dp[i] = dp[i - 1];
        } else {
            // 2.3
            dp[i] = i >= 2 ? dp[i - 1] + dp[i - 2] : dp[i - 1] + 1;
        }
    }
    return dp[len - 1];
}

    # 70.兑换零钱（一）

    /**
 * 兑换零钱（一）
 * @param {Array} arr 
 * @param {Number} aim 
 * @returns {Number}
 */
function minMoney(arr, aim) {
    // 动态规划
    // 初始化 dp 数组
    let dp = new Array(aim + 1).fill(Infinity)
    // 面额 0 需要 0 张货币兑换
    dp[0] = 0

    // 状态转移
    for (let i = 0; i <= aim; i++) {
        // 循环遍历面额
        for (let coin of arr) {
            // 遍历数组里面的货币
            if (i - coin >= 0) {
                // 1.面额值大于货币值时可以兑换
                // 2.dp[i - coin] + 1： 表示兑换一枚当前货币后剩下的面额接着兑换所需要的货币数目
                // 3.每次都考虑是否兑换当前货币
                dp[i] = Math.min(dp[i], dp[i - coin] + 1)
            }
        }
    }
    return dp[aim] === Infinity ? -1 : dp[aim]
}

    # 72.连续子数组的最大和

    /**
 * 连续子数组的最大和
 * @param {Array} arr 
 * @returns {Number}
 */
function findGreatestSumOfSubArray(arr) {
    // 动态规划，前面 i - 1 个子数组和为正 or 负
    // 初始化 dp 数组
    let dp = []
    dp[0] = arr[0]

    // 状态转移方程
    for (let i = 1; i < arr.length; i++) {
        dp[i] = Math.max(dp[i - 1] + arr[i], arr[i])
    }
    return Math.max(...dp)
}

    # 73.最长回文子串

    /**
 * 最长回文子串
 * @param {String} A 
 * @returns {Number}
 */
function getLongestPalindrome(A) {
    let res = 0
    let len = A.length
    if (len === 0 || len === 1) {
        return len
    }
    for (let i = 0; i < len; i++) {
        // 偶数字符串
        const evenStr = isPalindrome(A, i, i)
        // 奇数字符串
        const oddStr = isPalindrome(A, i, i + 1)
        res = Math.max(res, evenStr.length, oddStr.length)
    }
    return res
}

/**
 * 辅助函数判断字符串是否是回文字符串
 * @param {String} s 
 * @param {Number} left 
 * @param {Number} right 
 * @return {String}
 */
function isPalindrome(s, left, right) {
    // 左右指针，从 s[l] 和 s[r] 分别向两边扩散，找到最长回文字符串
    while (left >= 0 && right < s.length && s[left] === s[right]) {
        left--
        right++
    }
    return s.substring(left + 1, right - left - 1)
}

    # 74.数字字符串转化成IP地址

    /**
 * 数字字符串转化成 IP 地址
 * @param {String} s 
 * @returns {Array}
 */
function restoreIpAddresses(s) {
    let ans = []
    let path = []
    const dfs = (str, count) => {
        if (count === 4 && str === "") {
            ans.push(path.join("."))
            return
        }

        for (let i = 1; i <= 3 && i <= str.length; i++) {
            if (parseInt(str.substring(0, i)) <= 255) {
                if (i >= 2 && str[0] === '0') {
                    return
                }
                path.push(str.substring(0, i))
                dfs(str.substring(i, str.length), count + 1)
                path.pop()
            }
        }
    }
    dfs(s, 0)
    return ans
}

    # 75.编辑距离（一）

    /**
 * 编辑距离
 * @param {String} str1 
 * @param {String} str2 
 * @returns {Number}
 */
function editDistance(str1, str2) {
    // 初始化 dp 数组
    let dp = new Array(str1.length + 1).fill(0).map(() => {
        new Array(str2.length + 1).fill(0)
    })
    // dp[i][j] 表示从 str1 的前 i 位变化为 str2 的前 j 位所需要的次数
    // str1[i] === str2[j] => dp[i][j] = dp[i-1][j-1]
    // str1[i] !== str2[j] => dp[i][j] = min (dp[i-1][j-1], dp[i][j-1]) + 1
    // dp[i-1][j-1] 修改一个字符
    // dp[i][j-1] 删除一个字符
    // dp[i-1][j] 插入一个字符
    dp[0][0] = 0
    for (let i = 1; i < str1.length + 1; i++) {
        dp[i][0] = dp[i - 1][0] + 1
    }
    for (let j = 1; j < str2.length + 1; j++) {
        dp[0][j] = dp[0][j - 1] + 1
    }

    for (let i = 1; i < str1.length + 1; i++) {
        for (let j = 1; j < str2.length + 1; j++) {
            if (str1[i - 1] === str2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1]
            } else {
                dp[i][j] = Math.min(dp[i - 1][j - 1], dp[i][j - 1]) + 1
            }
        }
    }
    return dp[str1.length][str2.length]
}

    # 76.正则表达式匹配

    /**
 * 正则表达式匹配
 * @param {String} str 
 * @param {String} pattern 
 * @return {Boolean}
 */
function mathc(str, pattern) {
    const strLength = str.length 
    const patternLength = pattern.length 
    // 1.dp 下标的含义
    // dp[i][j] 表示从 str 首字母到 i, 以及从 pattern 首字母到 j 的子字符串是否匹配
    // 只要存在 i = strLength - 1, 有一个 j  即为 true

    // 2.初始化 dp 数组
    const dp = new Array(strLength + 1).fill(0).map(() => {
        new Array(patternLength + 1).fill(false)
    })
    dp[0][0] = true
    for (let i = 2; i <= patternLength; i++) {
        if (pattern[i - 1] === '*') {
            dp[0][i] = dp[0][i - 2]
        }
    }

    // 3.状态转移
    for (let i = 1; i <= strLength; i++) {
        for (let j = 1; j <= patternLength; j++) {
            if (pattern[j - 1] !== '*') {
                if (pattern[j - 1] === '.' || pattern[j - 1] === str[i - 1]) {
                    dp[i][j] = dp[i - 1][j - 1]
                }
            } else if (j >= 2) {
                if (pattern[j - 2] === "." || pattern[j - 2] === str[i - 1]) {
                    dp[i][j] = dp[i][j - 2] || dp[i - 1][j]
                } else {
                    dp[i][j] = dp[i][j - 2]
                }
            }
        }
    }

    // 4.返回值
    return dp[strLength][patternLength]
}

    # 77.最长的括号子串

    /**
 * 最长的括号子串
 * @param {String} s 
 * @return {Number} max
 */
function longestValidParentheses(s) {
    if (!s) {
        return 0
    }

    // 辅助栈存储“ ( ”的下标
    let len = s.length
    let stack = []
    let start = -1
    let max = 0
    for (let i = 0; i < len; i++) {
        if (s[i] === ')') {
            if (!stack.length) {
                // 此时辅助栈里面还没有左括号，从 start 到 i都是不合法的,重置start
                start = i
            } else {
                // 从辅助栈里面取一个“ （ ”进行匹配
                stack.pop()
                if (stack.length) {
                    // 辅助栈里面还有多余的 "("，合法的字符串为可以进行匹配的字符串
                    max = Math.max(max, i - stack[stack.length - 1])
                } else {
                    max = Math.max(max, i - start)
                }
            }
        } else {
            stack.push(i)
        }
    }
    return max
}

function rob(nums) {
    let dp = [],
        len = nums.length
    if (len === 0) {
        return 0
    }

    # 79.打家劫舍（二）

    /**
 * 打家劫舍（二）
 * @param {Array} nums
 * @return {Number} 
 */
function rob(nums) {
    // 动态规划
    let len = nums.length
    // 1.不偷第一家
    let dp1 = []
    dp1[0] = 0, dp1[1] = nums[1]
    // 状态转移
    for (let i = 2; i < len; i++) {
        dp1[i] = Math.max(dp1[i - 1], nums[i] + dp1[i - 2])
    }

    // 2.偷第一家不偷最后一家
    let dp2 = []
    dp2[0] = nums[0], dp2[1] = Math.max(nums[0], nums[1])
    for (let i = 2; i < len; i++) {
        if (i === len - 1) {
            dp2[i] = Math.max(dp2[i - 1], nums[i - 2])
        }
        dp2[i] = Math.max(dp2[i - 1], nums[i] + dp2[i - 2])
    }
    return dp1[len - 1] > dp2[len - 1] ? dp1[len - 1] : dp2[len - 1]
}

// 1.暴力枚举
function maxProfit(prices) {
    let profit = 0,
        len = prices.length
    // 买进价格，不要同一天买入卖出 (没有收益)
    for (let i = 0; i < len; i++) {
        // 卖出价格
        for (let j = i + 1; j < len; j++) {
            let soldValue = prices[j] - prices[i]
            // 更新可能的最大收益值
            profit = Math.max(profit, soldValue)
        }
    }
    return profit
}

// 2.贪心
function maxProfit2(prices) {
    let minBuyPrice = prices[0],
        profit = 0,
        len = prices.length
    for (let i = 1; i < len; i++) {
        minBuyPrice = Math.min(minBuyPrice, prices[i])
        profit = Math.max(profit, prices[i] - minBuyPrice)
    }
    return profit
}
