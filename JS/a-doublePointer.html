    # 87.合并两个有序的数组

    /**
 * 合并连个有序的数组
 * @param {Array} A 
 * @param {Number} m 数组 A 的元素数目
 * @param {Array} B 
 * @param {Number} n 数组 B 的元素数目
 */
function mergeB2A(A, m, B, n) {
    // 双指针
    let arr = new Array(m + n)
    let p1 = 0
    let p2 = 0
    let i = 0
    while (p1 < m && p2 < n) {
        arr[i++] = A[p1] < B[p2] ? A[p1++] : B[p2++]
    }
    while (p1 < m) {
        arr[i++] = A[p1++]
    }
    while (p2 < n) {
        arr[i++] = B[p2++]
    }
    for (i = 0; i < m + n; i++) {
        A[i] = arr[i]
    }
    return A
}
let A = [1, 2, 3, 7]
let B = [2, 6, 8, 9]
console.log(mergeB2A(A, 4, B, 4))

    # 88.判断是否是回文字符串

    /**
 * 判断是否是回文字符串
 * @param {String} str
 * @returns {Boolean} 
 */
function isPalindromicString(str) {
    // 双指针
    let len = str.length
    if (len === 0 || len === 1) {
        return true
    }
    let start = 0,
        end = len - 1
    while (start <= end) {
        if (str[start] === str[end]) {
            start++
            end--
        } else {
            return false
        }
    }
    return true
}

    # 89.合并区间

    // @ts-nocheck
/**
 * 合并区间
 * @param {Array} intervals 
 * @returns {Array}
 */
function mergeSection(intervals) {
    let res = []
    // 根据数组里面的 start 进行排序
    intervals.sort((a, b) => a.start - b.start)
    let i = 0
    // 保证循环比较所有数组元素
    while (i < intervals.length) {
        let left = intervals[i].start
        let right = intervals[i].end
        // 保证一个局部最小 left，最大 right
        while (i < intervals.length - 1 && right >= intervals[i + 1].start) {
            right = Math.max(right, intervals[i + 1].end)
            i++
        }
        res.push(new Interval(left, right))
        // 寻找下一个区间
        i++
    }
    return res
}

function Interval(a, b) {
    this.start = a || 0
    this.end == b || 0
}

    # 90.最小覆盖子串

    /**
 * 最小覆盖子串
 * @param {String} S 
 * @param {String} T 
 */
function minWindow(S, T) {
    // map 记录目标子串中需要的字符和相应的个数
    const map = new Map()
    let lent = T.length
    let lens = S.length
    for (let i = 0; i < lent; i++) {
        map.set(T[i], map.has(T[i]) ? map.get(T[i]) + 1 : 1)
    }
    // 双指针从左边开始找
    let len = map.size
    let res = ""
    let l = 0,
        r = 0
    while (r < lens) {
        // 取当前左边的第一个字符串，若在右边找到，右边 map 缩减一个
        const nowc = S[r]
        if (map.has(nowc)) {
            map.set(nowc, map.get(nowc) - 1)
            if (map.get(nowc) === 0) {
                len--
            }
        }
        // 找到所需的子串，与上次找到的进行比较，更短则更新，这时候考虑左指针的移动
        while (len === 0) {
            let newstr = S.slice(l, r + 1)
            if (!res || res.length > newstr.length) {
                res = newstr
            }
            let nowl = S[l]
            // 若左指针指向的字符是所需要的，需要将该字符的所需数量进行加一
            // 此时，右指针移动的时候也要找到对应的字符
            if (map.has(nowl)) {
                map.set(nowl, map.get(nowl) + 1)
                if (map.get(nowl) === 1) {
                    len++
                }
            }
            l++
        }
        r++
    }
    return res
}

let S = "XDOYEZODEYXNZ";
let T = "XYZ";
console.log(minWindow(S, T));

    /**
 * 反转字符串
 * @param {String} str 
 */
function solve(str) {
    let len = str.length
    if (len === 0 || len === 1) {
        return str
    }
    return str.split("").reverse().join("")
}

    # 92.最长无重复子数组

    /**
 * 最长无重复子数组
 * @param {*} arr 
 * @return {Number}
 */
function maxLength(arr) {
    // 记录最长无重复子数组的长度
    let max = 0
    // 保存临时的无重复子数组
    let str = []
    // 遍历数组每一个元素
    for (let i in str) {
        let index = str.indexOf(arr[i])
        // 如果此时 str 里面发现重复的元素，则去掉前面的数组，重新开始生成子数组
        if (index !== -1) {
            str.splice(0, str.indexOf(arr[i]) + 1)
        }
        // 更新子数组的长度
        str.push(arr[i])
        max = Math.max(str.length, max)
    }
    return max
}

    # 93.盛水最多的容器

    // 1.暴力枚举
/**
 * 盛水最多的容器
 * @param {Array} height 
 * @return {Number}
 */
function maxArea(height) {
    let res = 0
    let len = height.length
    if (len < 2) {
        return 0
    }
    for (let i = 0; i < len - 1; i++) {
        for (let j = i + 1; j < len - 1; j++) {
            let deepth = Math.min(height[i], height[j])
            res = Math.max(deepth * (j - i), res)
        }
    }
    return res
}

// 2.贪心:水桶短板效应
function maxArea2(height) {
    let len = height.length
    if (len < 2) {
        return 0
    }

    // 定义左右双指针
    let left = 0,
        right = len - 1
    let max = 0,
        area = 0
    while (left < right) {
        area = Math.min(height[left], height[right])
        max = Math.max(area, max)
        if (height[left] < height[right]) {
            left++
        } else {
            right--
        }
    }
    return max
}

    # 94.接雨水问题

    // 接雨水问题
/**
 * max water
 * @param arr int整型一维数组 the array
 * @return long长整型
 */

 function maxWater(arr) {
    let res = 0;
    if (arr.length < 3) {
        return 0;
    }
    // 双指针
    let left = 0,
        right = arr.length - 1;
    let maxL = 0,
        maxR = 0;
    while (left < right) {
        maxL = Math.max(maxL, arr[left]);
        maxR = Math.max(maxR, arr[right]);
        if (maxL < maxR) {
            res += maxL - arr[left++];
        } else {
            res += maxR - arr[right--];
        }
    }
    return res;
}
