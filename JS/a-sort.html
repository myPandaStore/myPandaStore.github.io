    # 1.插入排序

    /**
 * 当需要将一组数据按照一定的顺序排列时，可以使用排序算法。
 * 插入排序（Insertion Sort）是一种简单直观的排序算法，
 * 它的工作原理是通过构建有序序列，对于未排序数据，在已排序序列中从后向前扫描，找到相应位置并插入。
 * @param {Array} arr 无序数组
 * @returns {Array} 升序排列的数组
 */
function insertionSort(arr) {
    // arr[0] 为第一组有序序列
    for (let i = 1; i < arr.length; i++) {
        let current = arr[i]
        let j = i - 1
        while (j >= 0 && arr[j] > current) {
            arr[j + 1] = arr[j]
            j--
        }
        arr[j + 1] = current
    }
    return arr
}

    # 2.堆排序

    // 堆排序（Heap Sort）是一种基于堆的排序算法，是选择排序的一种改进。
// 堆排序将待排序的元素构造成一个二叉堆（大顶堆或小顶堆），然后依次取出堆顶元素（最大值或最小值），将剩余的元素重新构造成堆，
// 继续取出堆顶元素，重复此过程直到所有元素均被取出。

function heapSort(arr) {
    let n = arr.length

    // Build heap
    for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
        heapify(arr, n, i)
    }

    // One by one extract element from heap
    for (let i = n - 1; i > 0; i--) {
        // Move current root to end
        let temp = arr[0]
        arr[0] = arr[i]
        arr[i] = temp

        // Call max heapify on the reduced heap
        heapify(arr, i, 0)
    }
    return arr
}

function heapify(arr, n, i) {
    let largest = i; // Initialize largest ar root
    let left = 2 * i + 1
    let right = 2 * i + 2

    // If left child is larger than root
    if (left < n && arr[left] > arr[largest]) {
        largest = left
    }

    // If right child is larger than largest so far
    if (right < n && arr[right] > arr[largest]) {
        largest = right
    }

    // If largest is not root
    if (largest !== i) {
        let swap = arr[i]
        arr[i] = arr[largest]
        arr[largest] = swap
    }

    // Recursively heapify the affected sub-tree
    heapify(arr, n, largest)
}

    # 3.归并排序

    var arr = [6, 3, 7, 2, 5, 3]

//归并排序，分治
//生成左右两个有序子列
function mergeSort(arr) {
    var len = arr.length;
    if (len < 2) {
        return arr
    }
    var middle = Math.floor(len / 2),
        left = arr.slice(0, middle),
        right = arr.slice(middle);
    return merge(mergeSort(left), mergeSort(right))
}

//合并两个有序子列
function merge(left, right) {
    var result = [];
    while (left.length && right.length) {
        if (left[0] <= right[0]) {
            result.push(left.shift())
        } else {
            result.push(right.shift())
        }
    }
    while (left.length) {
        result.push(left.shift())
    }
    while (right.length) {
        result.push(right.shift())
    }
    return result
}

    # 4.快速排序

    function quickSort(arr) {
    let len = arr.length
    if (len === 0 || len === 1) return arr
    let leader = arr[0],left = [], right = []
    for (let i = 1; i < len; i++) {
        if (arr[i] < leader) {
            left.push(arr[i])
        } else {
            right.push(arr[i])
        }
    }
    left = quickSort(left)
    right = quickSort(right)
    return left.concat([leader], right)
}

    # 5.冒泡排序

    //两两比较，将最大放到末尾，然后将次大放到倒数第二位，循环  
var arr = [2, 3, 1, 5, 2, 6, 9];

function bubbleSort(arr) {
    let len = arr.length
    for (let i = len - 1; i >= 0; i--) {
        for (let j = 0; j < i; j++) {
            if (arr[j] > arr[i]) {
                let temp = arr[j]
                arr[j] = arr[i]
                arr[i] = temp
            }
        }
    }
    return arr
}

    # 6.希尔排序

    //https://www.cnblogs.com/chengxiao/p/6104371.html
//希尔排序 == 分组进行插入排序
function shellSort(arr) {
    for (let gap = (arr.length / 2); gap > 0; gap = (gap / 2)) {
        //从第gap个元素，逐个对其所在组进行直接插入排序
        for (let i = gap; i < arr.length; i++) {
            let j = i
            while (j - gap >= 0 && arr[j] < arr[j - gap]) {
                swap(arr, j, j - gap)
                j -= gap
            }
        }
    }
    return arr
}


function swap(arr, i, j) {
    if (arr[i] < arr[j]) {
        let temp = arr[i]
        arr[i] = arr[j]
        arr[j] = temp
    }
}

    # 7.选择排序

    //选择排序
//取出未排序的第一个元素，遍历该元素之后的部分并进行比较，第一次取第一个
//如果有更小的就更换位置

//最小 次小， 交换索引值 交换值

var arr = [3, 5, 2, 7, 4, 6, 8]
function selectSort(arr) {
    let len = arr.length
    for (let i = 0; i < len; i++) {
        let minIndex = i
        for (let j = i + 1; j < len; j++) {
            if (arr[j] < arr[minIndex]) {
                minIndex = j
            }
        }
        if (minIndex !== i) {
            let temp = arr[i]
            arr[i] = arr[minIndex]
            arr[minIndex] = temp
        }
    }
    return arr
}       
console.log(selectSort(arr))

    # 7.树形构建问题

    function buildTree(data, pid = 0) {  
    const result = [];  
    for (let item of data) {  
        if (item.pid === pid) {  
            const children = buildTree(data, item.id);  
            if (children.length) {  
                item.children = children;  
            }  
            result.push(item);  
        }  
    }  
    return result;  
}
