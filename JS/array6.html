    # 1.map

    Array.prototype.myMap = function(callback) {
    //处理数组类型异常
    if(this === null || this === undefined){
        throw new TypeError('Cannot read property "map" of null or undefined');
      }
      // 处理回调类型异常
    if(Object.prototype.toString.call(callback) !== '[object Function]'){
        throw new TypeError(callback + 'is not a function')
      }
    //this指向该数组
    var len = this.length
    var newArr = new Array(len)
    for (let i = 0; i < len; i++) {
        var value = this[i]
        if (value == undefined) continue
        var mappedValue = callback.call(null, value)
        newArr[i] = mappedValue
    }
    return newArr
}

    # 2.reduce

    Array.prototype.myReduce = function(callback, initValue) {
    //容错校验
      if (this == null || this == undefined) {
        throw new TypeError('Cannot read Property "reduce" of null or undefined')
      }
      if (Object.prototype.toString.call(callback) !== '[object Function]') {
        throw new TypeError(callback + 'is not a function')
      }
      var len = this.length
      //不传默认值会自动以第一个元素为初始值，然后从第二个元素开始依次累计。
      var k, accumlator
      if (initValue) {
          k = 0
          accumlator = initValue
      } else {
          k = 1
          accumlator = this[0]
      }
      for (var i = k; i < len; i++) {
          accumlator = callback.call(undefined, accumlator, this[i], this)
      }
      return accumlator
}

    Array.prototype.mySplice = function(startIndex, deleteCount, ...addElement) {
    //插入参数的个数
    let argumentsLen = arguments.length;
    let array = Object(this);

    let len = array.length;

    let deleteArr = new Array(deleteCount);

    //拷贝删除的元素
    sliceDeleteElements(array, startIndex, deleteCount, deleteArr);
    //移动删除元素后面的元素
    movePostElements(array, startIndex, len, deleteCount, addElements);
    //插入新元素
    for (let i = 0; i < addElements.length; i++) {
        array[startIndex + i] = addElements[i];
    }
    array.length = len - deleteCount + addElements.length;
    return deleteArr;
}

    # 4.filter

    Array.prototype.myFilter = function(callback) {
    //容错判断
    //1.数组类型不能为null 或 undefined
    if (this == null || this == undefined) {}
    //2.回调函数类型正确
    if (Object.prototype.toString.call(callback) === '[object Function]') {}

    var newArr = []
    var len = this.length
    for (let i = 0; i < len; i++) {
        if (callback[this[i]]) {
            newArr.push(this[i])
        }
    }
}

    Array.prototype.myPush = function(...args) {
    var len = args.length
    var arrLen = this.length
    if (len + arrLen > Math.pow(2, 32) - 1) {
        throw new TypeError('the number is over the limited')
    }
    for (let i = 0; i < len; i++) {
        this[arrLen + i] = args[i]
    }
    this.length = arrLen + len
    return this.length
}

    # 6.pop

    Array.prototype.myPop = function() {
    var len = this.length
    var a = this[len - 1]
    delete this[len - 1]
    this.length = len - 1
    return a
}

    //1.bubbleSort
////两两比较，将最大放到末尾，然后将次大放到倒数第二位，循环  
function bubbleSort(arr) {
    let len = arr.length
    for (let i = len - 1; i >= 0; i--) {
        for (let j = 0; j < i; j++) {
            if (arr[j] > arr[i]) {
                let temp = arr[j]
                arr[j] = arr[i]
                arr[i] = temp
            }
        }
    }
    return arr
}

     //2.归并排序，分治
 //生成左右两个有序子列
 function mergeSort(arr) {
     let len = arr.length
     if (len < 2) return arr
     let middle = Math.floor(len / 2),
        left = arr.slice(0, middle),
        right = arr.slice(middle)
     return merge(mergeSort(left), mergeSort(right))

     //合并两个有序子列
     const merge = (left, right) => {
         var res = []
         if (!left) return right
         if (!right) return left
         while (left.length && right.length) {
             if (left[0] < right[0]) {
                 res.push(left.shift())
             } else {
                 res.push(right.shift())
             }
         }
         while (left.length) res.push(left.shift())
         while (right.length) res.push(right.shift())
         return res
     }
 }

    //3.快排
function quickSort(arr) {
    if(arr.length < 2){
        return arr
    }
    let leader = arr[0],left = [],right = []
    for (let i = 1; i < arr.length; i++) {
        if (arr[i] < leader) left.push(arr[i])
        if (arr[i] > leader) right.push(arr[i])
    }
    left = quickSort(left)
    right = quickSort(right)
    return left.concat([leader], right)
}

    //4.插入排序
//1.首先有一个有序的序列，可以认为第一个元素就是已经排序的序列
//2.从未排序序列中取出一个元素，往有序序列中插入合适的位置
function insertSort(arr) {
    for (var i = 1; i < arr.length; i++) {
        let j = i - 1
        const curValue = arr[i]
        while (j >= 0 && arr[j] > curValue) {
            arr[j + 1] = arr[j]
            j--
        }
        arr[j + 1] = curValue
    }
    return arr
}

    //5.选择排序
//取出未排序的第一个元素，遍历该元素之后的部分并进行比较，第一次取第一个
//如果有更小的就更换位置
function selectSort(arr) {
    for (let i = 0; i < arr.length; i++) {
        let minIndex = i
        for (let j = i + 1; j < arr.length; j++) {
            if (arr[j] < arr[minIndex]) {
                minIndex = j
            }
        }
        if (minIndex !== i) {
            let temp = arr[i]
            arr[i] = arr[minIndex]
            arr[minIndex] = temp
        }
    }
    return arr
}

    function shellSort(arr) {
    //外层循环，gap分组，内层循环，组内直接插入排序
    for (let gap = Math.floor(arr.length / 2); gap > 0; gap = (gap / 2)) {
        for (let i = gap; i < arr.length; i++) {
            let j = i
            while (j - gap >= 0 && arr[j] < arr[j - gap]) {
                let temp = arr[j]
                arr[j] = arr[j - gap]
                arr[j - gap] = temp
                j -= gap
            }
        }
    }
    return arr
}
