    # 17.二分查找-I

    /**
 * 二分查找
 * @param {Array} nums 
 * @param {Number} target 
 * @returns {Number} 索引值
 */
function dichotomySearch(nums, target) {
    let len = nums.length
    let left = 0, right = len - 1
    while (left <= right) {
        let middle = Math.floor((left + right) / 2)
        if (nums[middle] === target) {
            return middle
        } else if (nums[middle] > target) {
            right = middle - 1
        } else {
            left = middle + 1
        }
    }
    return -1
}

    # 18.二维数组中的查找

    /**
 * 二维数组中的查找
 * @param {Number} target 目标值
 * @param {*} array 二维数组
 * @returns {Boolean}
 */
function findBetweenMatrix(target, array) {
    let m = array.length
    let n = array[0].length
    // 从左下角开始找
    let i = m - 1,
        j = 0
    while (i >= 0 && j < n) {
        if (target < array[i][j]) {
            i--
        } else if (target > array[i][j]) {
            j++
        } else {
            return true
        }
    }
    return false
}

    # 19.寻找峰值

    /**
 * 寻找峰值
 * @param {Array} nums 
 * @return {Number} 索引值
 */
function findPeakElement(nums) {
    // 二分查找
    let left = 0, right = nums.length - 1
    while (left < right) {
        let mid = Math.floor((right - left) / 2) + left
        if (nums[mid] < nums[mid + 1]) {
            left = mid + 1
        } else {
            right = mid
        }
    }
    return right
}

    # 20.数组中的逆序对

    /**
 * 数组中的逆序对
 * @param {Array} data 
 * @returns {Number}
 */
function inversPairs(data) {
    // 利用归并算法
    let dataLen = data.length
    let res = 0
    if (dataLen < 2) {
        return 0
    }
    const mergeSort = (start, end) => {
        if (start === end) {
            return data[start]
        }
        let mid = start + Math.floor((end - start) / 2)
        let leftChild = mergeSort(start, mid)
        let rightChild = mergeSort(mid)
        let arr = new Array(end - start + 1).fill(0)
        let left = 0, right = 0, now = 0
        while (left < leftChild.length && right < rightChild.length) {
            if (leftChild[left] <= rightChild[right]) {
                arr[now++] = leftChild[left++]
            } else {
                arr[now++] = rightChild[right++]
                res += leftChild.length - left
            }
        }
        while (left < leftChild.length) {
            arr[now++] = leftChild[left++]
        }
        while (right < rightChild.length) {
            arr[now++] = rightChild[right++]
        }
        return arr
    }
    mergeSort(0, dataLen - 1)
    return res % 1000000007
}

    # 21.旋转数组的最小数字

    /**
 * 旋转数组的最小数组
 * @param {Array} rotateArray 
 * @return {Number}
 */
function minNumberInRotateArray(rotateArray) {
    // 双指针
    let [left, right] = [0, rotateArray.length - 1]
    while (left < right) {
        let mid = left + Math.floor((right - left) / 2)
        if (rotateArray[mid] < rotateArray[right]) {
            right = mid
        } else if (rotateArray[mid] > rotateArray[right]) {
            left = mid + 1
        } else {
            right--
        }
    }
    return rotateArray[left]
}

    # 22.比较版本号

    /**
 * 比较版本号
 * @param {String} version1 
 * @param {String} version2 
 */
function compareVersion(version1, version2) {
    const vArr1 = version1.split(".")
    const vArr2 = version2.split(".")
    let i = 0
    // 将分割后的数组各项转化为数字即可解决前导 0 的问题
    while (i < vArr1.length || i < vArr2.length) {
        let x = 0
        let y = 0
        if (i < vArr1.length) {
            x = parseInt(vArr1[i])
        }
        if (i < vArr2.length) {
            y = parseInt(vArr2[i])
        }
        if (x > y) {
            return 1
        }
        if (x < y) {
            return -1
        }
        i++
    }
    return 0
}
