/**
 * 反转链表
 * @param {*} head 
 * @return {*} head
 */
function reverseList(head) {
    if (head === null || head.next === null) {
        return head
    }
    // 定义一个前驱节点
    let pre = null
    // 定义一个当前节点
    let cur = head
    while (cur) {
        // 定义一个后继节点
        let next = cur.next
        // 执行交换操作
        cur.next = pre
        pre = cur
        cur = next
    }
    return pre
}

    # 2.链表内指定区间反转

    /**
 * 链表内指定区间反转
 * @param {*} head 
 * @param {*} m 
 * @param {*} n 
 * @return {*} head
 */
function reverseBetween(head, m, n) {
    // 初始化虚拟节点
    let p = new ListNode('')
    p.next = head
    let pre = p

    // 找到区间前一个节点
    for (let i = 0; i < m - 1; i++) {
        pre = pre.next
    }

    // 找到区间终点
    let rightNode = pre
    for (let i = 0; i < n - m + 1; i++) {
        rightNode = rightNode.next
    }

    // 截取区间链表
    let leftNode = pre.next
    let cur = rightNode.next

    // 切断连接
    pre.next = null
    rightNode.next = null

    // 区间链表反抓
    reverseList(leftNode)

    // 连接回原来的链表
    pre.next = rightNode
    leftNode.next = cur

    return p.next

}

        cur.next = pre

        cur = next
    }
    return pre

    # 3.链表中的节点每k个一组反转

    /**
 * 链表中的节点每 k 个一组反转
 * @param {*} head 
 * @param {*} k 
 * @return {*} head
 */
function reverseKGroup(head, k) {
    if (!head) {
        return null
    }
    let start = head, end = head
    // 拿到 k 个一组的链表
    for (let i = 0; i < k; i++) {
        // 如果在还没有遍历到第 k 个，end 空了，即 head 链表个数不满足 k 个，直接返回原链表
        if (!end) {
            return head
        } else {
            end = end.next
        }
    }
    // 反转当前这组链表
    let newHead = reverseList(start) 
    // 反转后的链表连接下一组链表
    start.next = reverseKGroup(end, k)
    return newHead
}

    # 4.合并两个排序的链表

    /**
 * 合并两个排序的链表
 * @param {*} pHead1  链表 1 的头节点
 * @param {*} pHead2  链表 2 的头节点
 * @returns {*} 返回合并后的新的递增的链表
 */
function mergeTwoLists(pHead1, pHead2) {
    if (!pHead1) {
        return pHead2
    }
    if (!pHead2) {
        return pHead1
    }
    // 生成虚拟头节点
    let head = new ListNode('')
    // 保存当前虚拟头节点，以对其进行链表的相关操作
    let p = head
    // 保存两个排序链表的头部节点
    let p1 = pHead1,
        p2 = pHead2
    while (p1 && p2) {
        if (p1.val <= p2.val) {
            p.next = p1
            p = p.next
            p1 = p1.next
        } else {
            p.next = p2
            p = p.next
            p2 = p2.next
        }
    }
    while (p1) {
        p.next = p1
    }
    while (p2) {
        p.next = p2
    }
    return head.next
}

function mergeTwoLists2(pHead1, pHead2) {
    if (!pHead1) {
        return pHead2
    }
    if (!pHead2) {
        return pHead1
    }
    if (pHead1.val <= pHead2.val) {
        pHead1.next = mergeTwoLists2(pHead1.next, pHead2)
        return pHead1
    } else {
        pHead2.next = mergeTwoLists2(pHead1, pHead2.next)
        return pHead2
    }
}

    # 5.合并k个已经排序的链表

    /**
 * 合并 k 个已经排序的链表
 * @param {*} lists 
 * @return {*} head 合并后的链表头部节点
 */
function mergeKlists(lists) {
    let len = lists.length
    if (len === 0) {
        return null
    }
    if (len === 1) {
        return lists[0]
    }
    return mergeLists(lists, 0, len)
}

// 归并排序
function mergeLists(lists, start, end) {
    if (start === end) {
        // 归并边界 只剩一个元素 直接返回·
        return lists[start]
    }
    let mid = start + (end - start) / 2 // 分治中点
    let leftList = mergeLists(lists, start, mid) // 递归得到左边有序链表
    let rightList = mergeLists(lists, mid + 1, end) // 递归得到右边有序链表
    return merge(leftList, rightList) // 合并左右两个有序链表
}

// 合并两个有序链表
function merge(head1, head2) {
    if (!head1) {
        return head2
    }
    if (!head2) {
        return head1
    }
    if (head1.val <= head2.val) {
        head1.next = merge(head1.next, head2)
        return head1
    } else {
        head2.next = merge(head1, head2.next)
        return head2
    }
}

    # 6.判断链表中是否有环

    /**
 * 判断链表中是否有环
 * @param {*} head 
 * @return {Boolean}
 */

// 1.利用 set
function hasCycle(head) {
    if (!head) {
        return false
    }
    let set = new Set()
    let p = head
    while (p) {
        if (set.has(p)) {
            return true
        }
        set.add(p)
        p = p.next
    }
    return false
}

// 2.快慢指针
function hasCycle2(head) {
    let fast = head,
        slow = head
    //  超过三个节点，链表才能成环
    if (fast === null || fast.next === null || fast.next === null) {
        return false
    }
    while (fast && fast.next) {
        fast = fast.next.next
        slow = slow.next
        if (fast === slow) {
            return true
        }
        return false
    }
}

    /**
 * 链表中倒数最后 k 个节点
 * @param {*} pHead 
 * @param {*} k 
 * @returns {*} head    
 */
function findKthToTail(pHead, k) {
    // 快慢指针
    let fast = pHead,
        slow = pHead
    for (let i = 0; i < k; i++) {
        if (fast === null) {
            return null
        }
        fast = fast.next
    }
    while (fast !== null) {
        fast = fast.next
        slow = slow.next
    }
    return slow
}

    /**
 * 删除链表的倒数第 n 个节点
 * @param {*} head 
 * @param {*} n 
 * @return {*} head
 */
function removeNthFromEnd(head, n) {
    // 快慢指针
    let fast = head,
        slow = head
    for (let i = 0; i < n; i++) {
        fast = fast.next
    }
    if (fast === null) {
        return head.next
    }
    let pre
    while (fast !== null) {
        fast = fast.next
        pre = slow
        slow = slow.next
    }
    pre.next = slow.next
    return head
}

    /**
 * 两个链表的第一个公共节点
 * @param {*} pHead1 
 * @param {*} pHead2 
 */
function findFirstCommonNode(pHead1, pHead2) {
    // 模拟将两个链表连接起来找第一次出现的公共节点
    if (!pHead1 || !pHead2) {
        return null
    }
    let p1 = pHead1,
        p2 = pHead2
    while (p1 !== p2) {
        p1 = (p1 === null ? pHead2 : p1.next)
        p1 = (p1 === null ? pHead2 : p1.next)
    }
    return p1
}

    # 11.链表相加

    /**
 * 链表相加（二）
 * @param {*} head1 
 * @param {*} head2 
 * @return {*} head
 */
function addInList(head1, head2) {
    // 将两个链表倒序
    let reHead1 = reverse(head1)
    let reHead2 = reverse(head2)

    let head = new ListNode(0)
    let sumHead = head

    // 是否有进位
    let exceed = 0
    // 循环相加倒序后的链表（从个位开始加）
    while (reHead1 || reHead2) {
        sumHead.next = new ListNode(0)
        const val1 = parseInt(reHead1 ? reHead1.val : 0)
        const val2 = parseInt(reHead2 ? reHead2.val : 0)
        const sum = val1 + val2 + exceed
        if (sum >= 10) {
            sumHead.next.val = sum % 10
            exceed = 1
        } else {
            sumHead.next.val = sum % 10
            exceed = 0
        }
        reHead1 = reHead1 ? reHead1.next : null
        reHead2 = reHead2 ? reHead2.next : null
        sumHead = sumHead.next
    }
    // 加到最后一位如果还有进位就在加一个进位
    if (exceed) {
        sumHead.next = new ListNode(exceed)
    }

    // 最后将新链表反转
    const reSum = reverse(head.next)
    return reSum
}

function reverse(head) {
    let pre = null
    let cur = head
    while (cur) {
        let next = cur.next
        cur.next = pre
        pre = cur
        cur = next
    }
    return pre
}

    # 12.单链表的排序

    /**
 * 单链表的排序
 * @param {*} head
 * @return {*} head 
 */
function sortInList(head) {
    if (!head || !head.next) {
        return head
    }
    let arr = []
    let cur = head
    while (cur) {
        arr.push(cur.val)
        cur = cur.next
    }
    arr.sort((a, b) => a - b)
    cur = head
    for (let i = 0; i < arr.length; i++) {
        cur.val = arr[i]
        cur = cur.next
    }
    return head
}

    # 13.判断一个链表是否是回文结构

    /**
 * 判断一个链表是否是回文结构
 * @param {*} head 
 * @return {Boolean}

// 1.反转链表
function isPalindrome(head) {
    let reHead = reverse(head)
    while (head && reHead) {
        if (head.val !== reHead.val) {
            return false
        }
        head = head.next
        reHead = reHead.next
    }
    return true
}

function reverse(head) {
    if (!head || !head.next) {
        return head
    }
    let pre = null
    let cur = head
    while (cur) {
        let next = cur.next
        cur.next = pre
        pre = cur
        cur = next
    }
    return pre
}

// 2.收集节点值
function isPalindrome2(head) {
    let valArr = []
    while (head) {
        valArr.push(head.val)
        head = head.next
    }
    for (let i = 0, j = valArr.length - 1; i < j; i++, j--) {
        if (valArr[i] !== valArr[j]) {
            return false
        }
    }
    return true
}

    # 14.链表的奇偶重排

    /**
 * 链表的奇偶重排
 * @param {*} head 
 * @return {*} head
 */
function oddEvenList(head) {
    // 1.分成两个链表
    let head_odd = new ListNode(-1)
    let head_even = new ListNode(-1)
    // 2.保存奇数链表和偶数链表的头部
    let last_odd = head_odd
    let last_even = head_even
    // 3.用一个变量标记奇偶
    let i = 1
    // 4.遍历链表，分别记住头尾
    while (head) {
        if (i % 2 === 1) {
            last_odd.next = head
            last_odd = head
        } else {
            last_even.next = head
            last_even = head
        }
        head = head.next
        i++
    }
    // 偶数链表尾部的 next 重置为 null
    last_even.next = null
    // 连接奇数链表尾部和偶数链表头部
    last_odd.next = head_even.next
    return head_odd.next
}
